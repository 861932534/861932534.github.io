Subsequences Summing to Sevens S（洛谷）

方法一：暴力前缀和

#include<bits/stdc++.h>
using namespace std;
const int N=5e4+10;
int a[N];
int sum[N];
int n;
int main()
{
    cin>>n;
    for(int i=1;i<=n;i++){
        cin>>a[i];
        sum[i]=sum[i-1]+a[i];
    }
    int ans=0,m=n;
    while(m--){
        for(int i=1;i<=n-m;i++){
            if((sum[i+m]-sum[i-1])%7==0){
                ans=m;
                cout<<ans+1<<endl;
                return 0;
            }
        }
    }
    cout<<0<<endl;
    return 0;
}
//部分数据TLE，时间复杂度为（N*N）

方法二：前缀和+同余定理 题解

#include<bits/stdc++.h>
using namespace std;
const int N=5e4+10;
int n;
int pre[N];
int first[7],last[7];
int main()
{
    cin>>n;
    for(int i=1;i<=n;i++){
        cin>>pre[i];
        pre[i]=(pre[i-1]+pre[i])%7;
    }
    for(int i=n;i>=1;i--)
        first[pre[i]]=i;
    first[0]=0;//注意这一步很关键 
    for(int j=1;j<=n;j++)
        last[pre[j]]=j;
    int ma=0;
    for(int i=0;i<=6;i++)
        ma=max(ma,last[i]-first[i]);
    cout<<ma<<endl;
    return 0;
}

方法三：动态规划（有一亿点难理解）

#include<bits/stdc++.h>
using namespace std;
int n,f[50005][7],ans;
main()
{
    cin>>n;
    for(int i=1,x;i<=n;i++)
    {
        cin>>x;
        for(int j=0;j<7;j++)
            if(f[i-1][j])
                f[i][(j+x)%7]=f[i-1][j]+1;
        f[i][x%7]=max(f[i][x%7],1);
    }
    for(int i=1;i<=n;i++)
        ans=max(ans,f[i][0]);
    printf("%d\n",ans);
    return 0;
}
