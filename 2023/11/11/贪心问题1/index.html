二、陶陶摘苹果（升级版）—-贪心问题

方法1（我的解法）：

思路：使用pair来存储高度和体力值，用sort两次排序。第一次按高度从小到大排序（排序范围：1-n），第二次在第一次的基础上按体力值从小到大排序（排序范围1-cnt//cnt是小于等于淘淘能够够到的高度的苹果的数量）。然后用循环扫一遍就可以。

代码：

#include<bits/stdc++.h>
using namespace std;
const int N=1e4;
typedef pair<int,int> PII;
PII x[N];
int n,s,a,b;
bool cmp(PII a,PII b){
    return a.second<b.second;
}
int main()
{
    cin>>n>>s>>a>>b;
    int cnt=0;
    for(int i=0;i<n;i++){
        cin>>x[i].first>>x[i].second;
        if(x[i].first<=a+b) cnt++;
    }
    sort(x,x+n);
    sort(x,x+cnt,cmp);
    int res=0;
    for(int i=0;i<cnt;i++){
        if(s>=x[i].second){
            res++;
            s-=x[i].second;
            //cout<<x[i].first<<x[i].second<<endl;
        }
    }
    cout<<res<<endl;	
    return 0;
}

方法2：使用dfs，搭配剪枝，使用变量存储来减少函数的调用以及记忆化搜索来降低时间复杂度。

代码：

#include<bits/stdc++.h>

using namespace std;
int n,s,a,b;
bool visit[5005][5005];//记忆化搜索
int mem[5005][5005];
struct apple {
    int xi,yi;
}ap[5005];
int dfs(int num,int rest){
    if(num>n||a+b<ap[num].xi) return 0;//剪枝 
    if(visit[num][rest]) return mem[num][rest];
    visit[num][rest]=true;
    int maxn=dfs(num+1,rest);//减少函数的调用 
    if(ap[num].xi<=a+b&&rest>=ap[num].yi){
        int t= dfs(num+1,rest-ap[num].yi)+1;//减少函数的调用
        maxn=t>maxn?t:maxn;
    }
    return mem[num][rest]=maxn; 
} 	

bool cmp(apple a,apple b){//从小到大 
    return a.xi<b.xi;
}
int main()
{
    cin>>n>>s>>a>>b;//苹果个数，体力值，板凳高度，伸长手高度 
    for(int i=1;i<=n;i++) cin>>ap[i].xi>>ap[i].yi;
    sort(ap+1,ap+n+1,cmp);
    cout<<dfs(1,s);	//面对第一个苹果，体力值为s 
    return 0;
}

方法3、dp（动态规划），比较简单的01背包问题

代码：

#include<bits/stdc++.h>

using namespace std;
int n,s,a,b;
int xi[5005],yi[5005];
int dp[5005][1001];
int main()
{
    cin>>n>>s>>a>>b;
    for(int i=1;i<=n;i++)
        cin>>xi[i]>>yi[i];
    for(int i=1;i<=n;i++){
        for(int j=0;j<=s;j++){	//j为什么是从零开始的，因为体力可以为零 
            dp[i][j]=dp[i-1][j];
            if(xi[i]<=a+b&&j>=yi[i])
            //dp[i][j]=max(dp[i-1][j-yi[i]]+1,dp[i][j]);max函数很慢，使用三目运算比较快 
            dp[i][j]=dp[i-1][j-yi[i]]+1>dp[i][j]?dp[i-1][j-yi[i]]+1:dp[i][j];
            //状态转移方程
        }
    }
    cout<<dp[n][s]<<endl;
    return 0;
}

方法4、桶排序 知识点： ( 1 2 ) 视频：（1）

代码：

#include <bits/stdc++.h>
using namespace std;
const int N=5e3+10;
int n,s,a,b,t;
int c[101];
int main() {
    cin>>n>>s>>a>>b;
    a+=b;
    for(int i=0;i<n;i++){
        int x,y;
        cin>>x>>y;
        if(x<=a) c[y]++;
    }
    for(int i=0;i<=100&&s>=0;i++){//有的体力值是零 
        while(c[i]){
            s-=i;
            t++;
            c[i]--;
            if(s<0){
                t--;
                break;
            }
        }
    }
    cout<<t<<endl;
    return 0;
}
